## 2. P0 - 阻塞性问题

### P0-1: BullMQ Worker 定义了但从未启动

**问题位置**：`apps/ai-video-api/src/modules/queue/queue.module.ts`、`workers/image.worker.ts`、`workers/video.worker.ts`

**现状**：`ImageWorker` 和 `VideoWorker` 类已定义，但 `QueueModule` 只注册了 `QueueService`，没有任何地方实例化 Worker。这意味着任务被加入 BullMQ 队列后**无人消费**，生成功能实际无法运转。

**影响**：所有图片/视频生成任务永久停留在 `queued` 状态，核心功能不可用。

**方案**：在 `QueueModule` 中注册 Worker 并在应用生命周期中启动。以下为可直接粘贴的模块改造代码（需替换现有 `queue.module.ts` 内容）：

```typescript
// apps/ai-video-api/src/modules/queue/queue.module.ts
import { Module, Global, OnModuleInit, OnModuleDestroy, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { QueueService } from './queue.service';
import { ImageWorker } from './workers/image.worker';
import { VideoWorker } from './workers/video.worker';
import { PrismaService } from '../../prisma/prisma.service';
import { ModelsService } from '../models/models.service';
import { AdapterFactory } from '../../common/adapters/adapter.factory';
import { StorageService } from '../storage/storage.service';
import { TaskGateway } from '../websocket/websocket.gateway';

@Global()
@Module({
  providers: [QueueService],
  exports: [QueueService],
})
export class QueueModule implements OnModuleInit, OnModuleDestroy {
  private readonly logger = new Logger(QueueModule.name);
  private imageWorker: ImageWorker;
  private videoWorker: VideoWorker;

  constructor(
    private readonly configService: ConfigService,
    private readonly prisma: PrismaService,
    private readonly modelsService: ModelsService,
    private readonly adapterFactory: AdapterFactory,
    private readonly storageService: StorageService,
    private readonly taskGateway: TaskGateway,
  ) {}

  async onModuleInit() {
    const redisUrl = this.configService.get<string>('REDIS_URL', 'redis://localhost:6379');

    this.imageWorker = new ImageWorker(redisUrl, async (data) => {
      await this.processJob(data);
    });

    this.videoWorker = new VideoWorker(redisUrl, async (data) => {
      await this.processJob(data);
    });

    this.logger.log('BullMQ workers started (image x3, video x2)');
  }

  private async processJob(data: any) {
    const { taskId, capability, modelId, apiKeyId, projectId } = data;

    // 1. 标记任务为 processing
    await this.prisma.generationTask.update({
      where: { id: taskId },
      data: { status: 'processing', startedAt: new Date() },
    });

    try {
      // 2. 解密 API Key
      const apiKey = await this.modelsService.getDecryptedApiKey(
        data.input?.userId || '', apiKeyId,
      );

      // 3. 根据 capability 调用对应 Adapter
      let resultUrl: string;
      if (capability === 'image') {
        const adapter = this.adapterFactory.getImageAdapter(modelId);
        resultUrl = (await adapter.generateImage(data.input, { apiKey })).url;
      } else if (capability === 'video') {
        const adapter = this.adapterFactory.getVideoAdapter(modelId);
        resultUrl = (await adapter.generateVideo(data.input, { apiKey })).url;
      } else {
        const adapter = this.adapterFactory.getTTSAdapter(modelId);
        resultUrl = (await adapter.synthesize(data.input, { apiKey })).url;
      }

      // 4. 写回结果
      await this.prisma.generationTask.update({
        where: { id: taskId },
        data: {
          status: 'completed', progress: 100,
          result: { url: resultUrl }, completedAt: new Date(),
        },
      });

      // 5. WebSocket 推送完成事件
      this.taskGateway.emitTaskProgress(data.input?.userId, {
        taskId, projectId, status: 'completed',
        progress: 100, resultUrl,
      });
    } catch (error: any) {
      await this.prisma.generationTask.update({
        where: { id: taskId },
        data: { status: 'failed', error: { message: error.message } },
      });
      throw error; // 让 BullMQ 重试
    }
  }

  async onModuleDestroy() {
    await this.imageWorker?.close();
    await this.videoWorker?.close();
    this.logger.log('BullMQ workers closed');
  }
}
```

> 注意：Worker 处理函数中需要 `userId` 来解密 Key，建议在 `QueueService.addGenerationJob` 时将 `userId` 一并放入 job data。

### P0-2: Compose 合成服务仅返回占位数据

**问题位置**：`apps/ai-video-api/src/modules/compose/compose.service.ts`

**现状**：`composeProject` 方法只返回一个「合成计划」对象，注释写着 "In production, this would call FFmpeg"，但没有实际调用 FFmpeg。README 宣称的「FFmpeg 合成最终视频」功能未实现。

**方案**：引入 `fluent-ffmpeg` 实现真正的视频拼接。以下为增强后的合成逻辑：

```typescript
// 需安装: pnpm add fluent-ffmpeg @types/fluent-ffmpeg
import ffmpeg from 'fluent-ffmpeg';
import { join } from 'path';
import { readFile } from 'fs/promises';

async composeVideo(userId: string, projectId: string) {
  const project = await this.prisma.project.findFirst({
    where: { id: projectId, userId },
    include: { shots: { orderBy: { sequence: 'asc' } } },
  });
  if (!project) throw new Error('项目不存在');

  const tempDir = join(process.env.TEMP_DIR || '/tmp', `compose-${projectId}`);
  const localFiles: string[] = [];

  // 1. 下载所有 shot 素材并转码为统一格式
  for (const shot of project.shots) {
    if (!shot.resultUrl) continue;
    const localPath = join(tempDir, `shot-${shot.sequence}.mp4`);
    await this.downloadAndConvert(shot.resultUrl, localPath, shot.duration);
    localFiles.push(localPath);
  }

  // 2. FFmpeg concat 拼接
  const outputPath = join(tempDir, `output-${Date.now()}.mp4`);
  await this.concatVideos(localFiles, outputPath);

  // 3. 上传到 MinIO
  const key = this.storageService.generateKey(projectId, 'compose');
  const buffer = await readFile(outputPath);
  const url = await this.storageService.uploadBuffer(key, buffer, 'video/mp4');

  // 4. 清理临时文件 & 更新项目状态
  await this.prisma.project.update({
    where: { id: projectId },
    data: { status: 'completed' },
  });

  return { data: { projectId, outputUrl: url, status: 'completed' } };
}

private concatVideos(inputs: string[], output: string): Promise<void> {
  return new Promise((resolve, reject) => {
    const command = ffmpeg();
    inputs.forEach((f) => command.input(f));
    command
      .on('end', () => resolve())
      .on('error', (err) => reject(err))
      .mergeToFile(output);
  });
}
```

## 3. P1 - 安全性增强

### P1-1: API Key 加密密钥与 JWT_SECRET 复用（高危）

**问题位置**：`apps/ai-video-api/src/modules/models/models.service.ts` 加密/解密方法

**现状**：`encryptApiKey` / `decryptApiKey` 直接使用 `JWT_SECRET` 作为 AES-256-GCM 的密钥来源：

```typescript
const secret = this.configService.get<string>('JWT_SECRET', 'your-jwt-secret');
const key = crypto.scryptSync(secret, 'ai-video-salt', 32);
```

**风险**：
1. JWT 密钥和加密密钥职责不同，混用违反密钥分离原则。JWT 签名可离线爆破，一旦泄露，所有用户的 AI API Key 全部可解密。
2. `scryptSync` 在每次加/解密时都重新派生密钥（代价高昂的 KDF），造成不必要的 CPU 开销。
3. salt 硬编码为 `'ai-video-salt'`，且 `.env.example` 默认值为 `'your-jwt-secret'`，生产环境若忘记修改则形同明文。

**方案**：引入独立的 `ENCRYPTION_KEY` 环境变量，启动时派生一次缓存复用：

```bash
# .env.example 追加:
# API Key 加密密钥（32 位以上随机字符串，与 JWT_SECRET 不同）
ENCRYPTION_KEY="generate-a-secure-random-string-here"
```

```typescript
// models.service.ts 改造:
private readonly encryptionKey: Buffer;

constructor(...) {
  const raw = this.configService.get<string>('ENCRYPTION_KEY');
  if (!raw || raw.length < 16) {
    throw new Error('ENCRYPTION_KEY 必须配置且长度不少于 16 字符');
  }
  // 启动时派生一次，后续复用，避免每次加解密都跑 scrypt
  this.encryptionKey = crypto.scryptSync(raw, 'ai-video-apikey-salt', 32);
}

private encryptApiKey(apiKey: string): string {
  const iv = crypto.randomBytes(16);
  const cipher = crypto.createCipheriv('aes-256-gcm', this.encryptionKey, iv);
  let encrypted = cipher.update(apiKey, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  const authTag = cipher.getAuthTag().toString('hex');
  return `${iv.toString('hex')}:${authTag}:${encrypted}`;
}
```

> 迁移注意：已有数据库中的加密数据需要用旧密钥解密后再用新密钥重新加密，建议写一次性迁移脚本。

### P1-3: JWT 无刷新机制，Token 过期即掉线

**问题位置**：`apps/ai-video-api/src/modules/auth/auth.service.ts`

**现状**：JWT 有效期 7 天，过期后用户需要重新登录。没有 refresh token 机制，长期使用体验差。

**方案**：引入双 Token 机制（access token 短期 + refresh token 长期）：

```typescript
// auth.service.ts 改造:
private signTokens(userId: string, email: string) {
  const accessToken = this.jwtService.sign(
    { sub: userId, email },
    { expiresIn: '15m' },
  );
  const refreshToken = this.jwtService.sign(
    { sub: userId, email, type: 'refresh' },
    { expiresIn: '7d' },
  );
  return { accessToken, refreshToken };
}

// 新增 refresh 方法:
async refresh(refreshToken: string) {
  const payload = this.jwtService.verify(refreshToken);
  if (payload.type !== 'refresh') {
    throw new UnauthorizedException('无效的 refresh token');
  }
  return this.signTokens(payload.sub, payload.email);
}
```

```typescript
// auth.controller.ts 新增端点:
@Post('refresh')
@ApiOperation({ summary: '刷新 Token' })
async refresh(@Body('refreshToken') token: string) {
  return this.authService.refresh(token);
}
```

## 4. P2 - 架构与可靠性

### P2-1: 分镜生成无数据库事务，中途失败丢数据

**问题位置**：`apps/ai-video-api/src/modules/storyboard/storyboard.service.ts`

**现状**：`generateShots` 先 `deleteMany` 删除旧分镜，再用 `Promise.all` 并行创建新分镜。如果某个 `create` 失败，旧分镜已删、新分镜只创建了一部分，数据丢失。

**方案**：用 `prisma.$transaction` 包裹删除+创建，保证原子性：

```typescript
// storyboard.service.ts generateShots 改造:
const shots = await this.prisma.$transaction(async (tx) => {
  await tx.shot.deleteMany({ where: { projectId } });
  const storyboard = await this.getOrCreateStoryboard(projectId);

  // 顺序创建而非 Promise.all，事务内并行无收益反而增加死锁风险
  const created: any[] = [];
  for (let index = 0; index < shotsData.length; index++) {
    const shot = shotsData[index];
    const record = await tx.shot.create({
      data: {
        projectId,
        storyboardId: storyboard.id,
        sequence: index + 1,
        prompt: shot.prompt || '',
        negativePrompt: shot.negativePrompt || '',
        duration: shot.duration || 3000,
        params: {
          description: shot.description || '',
          title: shot.title || `分镜 ${index + 1}`,
          characters: shot.characters || [],
          scene: shot.scene || '',
          emotion: shot.emotion || '',
          dialogue: shot.dialogue || '',
          narration: shot.narration || '',
          subtitle: shot.subtitle || '',
          camera: shot.camera || {},
        },
        status: 'draft',
      },
    });
    created.push(record);
  }
  return created;
});
```

### P2-2: 应用无优雅关停（SIGTERM 处理缺失）

**问题位置**：`apps/ai-video-api/src/main.ts`

**现状**：进程收到 SIGTERM 时直接退出，BullMQ 队列连接、数据库连接、WebSocket 连接都没有正常关闭，可能导致正在处理的任务丢失。

**方案**：

```typescript
// main.ts 追加:
async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  // 优雅关停：等待活跃连接完成
  app.enableShutdownHooks(); // 自动触发 onModuleDestroy / onApplicationShutdown

  await app.listen(port);

  process.on('SIGTERM', async () => {
    console.log('SIGTERM received, closing gracefully...');
    await app.close();
    process.exit(0);
  });
}
```

> `enableShutdownHooks()` 会让 NestJS 自动调用各模块的 `onModuleDestroy`，配合 P0-1 中 Worker 的 `close()` 方法即可实现完整关停。

### P2-6: 生成任务无重试上限保护与并发控制

**问题位置**：`apps/ai-video-api/src/modules/generations/generations.service.ts`

**现状**：BullMQ 配置了 `attempts: 3`，但 `GenerationTask` 的 `retryCount` 字段虽已建表却未在业务中使用。一个持续失败的任务会重试 3 次后进入 failed，但缺乏手动重试 API 和对单用户并发任务数的限制。

**方案**：

```typescript
// generations.controller.ts 新增重试端点:
@Post(':taskId/retry')
@ApiOperation({ summary: '重试失败的任务' })
async retryTask(@Param('projectId') projectId: string, @Param('taskId') taskId: string) {
  return this.generationsService.retryTask(projectId, taskId);
}

// generations.service.ts 新增:
async retryTask(userId: string, projectId: string, taskId: string) {
  const task = await this.prisma.generationTask.findFirst({
    where: { id: taskId, projectId },
  });
  if (!task) throw new NotFoundException('任务不存在');
  if (task.status !== 'failed') {
    throw new BadRequestException('只有失败的任务可以重试');
  }

  // 重新入队
  await this.prisma.generationTask.update({
    where: { id: taskId },
    data: { status: 'queued', retryCount: { increment: 1 }, error: null },
  });

  await this.queueService.addGenerationJob({
    taskId: task.id,
    projectId,
    shotId: task.shotId || undefined,
    capability: task.capability as any,
    modelId: task.modelId,
    apiKeyId: task.apiKeyId,
    input: { ...task.input as any, userId },
  });

  return { data: { taskId, status: 'queued' } };
}
```

## 5. P3 - 性能与体验优化

### P3-1: AI 模型列表无缓存，每次请求都查库

**问题位置**：`apps/ai-video-api/src/modules/models/models.service.ts`

**现状**：`listModels` 每次都直接查数据库。AI 模型列表是准静态数据（只在 seed 或后台更新时变化），频繁查库不必要。

**方案**：用内存缓存或 Redis 缓存：

```typescript
// 简单方案：模块级内存缓存 + TTL
private modelsCache: { data: any[]; expiry: number } | null = null;
private readonly CACHE_TTL = 5 * 60 * 1000; // 5 分钟

async listModels(capability?: string) {
  const now = Date.now();
  if (!this.modelsCache || this.modelsCache.expiry < now) {
    const models = await this.prisma.aIModel.findMany({
      orderBy: [{ capability: 'asc' }, { name: 'asc' }],
    });
    this.modelsCache = { data: models, expiry: now + this.CACHE_TTL };
  }

  const filtered = capability
    ? this.modelsCache.data.filter((m) => m.capability === capability)
    : this.modelsCache.data;

  return { data: filtered };
}

// 提供 invalidate 方法，供模型更新时调用
invalidateModelsCache() {
  this.modelsCache = null;
}
```

### P3-2: 列表接口无分页

**问题位置**：`projects.service.ts`、`generations.service.ts`、`models.service.ts`

**现状**：`listProjects`、`listTasks`、`listModels` 都用 `findMany` 全量返回。数据量增长后响应变慢、内存占用上升。

**方案**：统一分页封装：

```typescript
// 新建 src/common/pagination.dto.ts
import { IsOptional, IsInt, Min, Max } from 'class-validator';
import { Type } from 'class-transformer';

export class PaginationDto {
  @IsOptional() @Type(() => Number) @IsInt() @Min(1)
  page: number = 1;

  @IsOptional() @Type(() => Number) @IsInt() @Min(1) @Max(100)
  pageSize: number = 20;
}

// service 中使用:
async listProjects(userId: string, query: PaginationDto) {
  const { page, pageSize } = query;
  const [items, total] = await Promise.all([
    this.prisma.project.findMany({
      where: { userId },
      orderBy: { createdAt: 'desc' },
      skip: (page - 1) * pageSize,
      take: pageSize,
    }),
    this.prisma.project.count({ where: { userId } }),
  ]);
  return { data: items, pagination: { page, pageSize, total } };
}
```

### P3-4: 前端 Token 存储在 localStorage（XSS 风险）

**问题位置**：`apps/ai-video-web/lib/api.ts`

**现状**：accessToken 存在 `localStorage`，任何 XSS 攻击都能读取。虽然 Next.js 客户端渲染场景下 httpOnly cookie 有局限，但仍可优化。

**方案**：配合 P1-3 的 refresh token 机制，将 refresh token 存入 httpOnly cookie，access token 存内存（React state），减少暴露面：

```typescript
// 前端 AuthProvider 中维护 accessToken 状态
const [accessToken, setAccessToken] = useState<string | null>(null);

// 登录时：access token 存内存，refresh token 由后端 Set-Cookie 写入 httpOnly cookie
// 过期时：自动调用 /auth/refresh，cookie 自动携带 refresh token

// api.ts 改为动态注入:
api.interceptors.request.use((config) => {
  if (accessToken) {
    config.headers.Authorization = `Bearer ${accessToken}`;
  }
  return config;
});

// 401 时自动刷新:
api.interceptors.response.use(null, async (error) => {
  if (error.response?.status === 401 && !error.config._retry) {
    error.config._retry = true;
    const { data } = await api.post('/auth/refresh');
    setAccessToken(data.accessToken);
    error.config.headers.Authorization = `Bearer ${data.accessToken}`;
    return api(error.config);
  }
  return Promise.reject(error);
});
```

```typescript
          userId: (req as any).user?.id,
        }),
      },
    }),
    // ...其他模块
  ],
})
```

```typescript
// main.ts 中用 pino 替代默认 Logger:
const app = await NestFactory.create(AppModule, { bufferLogs: true });
```
